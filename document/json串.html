登录窗口： 客户端发送格式：{ ope:1, //操作指令 id:abc, pw:123456, } 服务器回复格式： 若登录成功： { ope:1, status:1, you:{ //你的个人信息 nick:, avatar:, id:, }， invite:[ //好友申请的数组 { from: //数字表示是你的申请还是他人对你申请 //0他人对你，1你对他人 id: nick: avatar:1 //数字代表固定头像 status: //0未审批，1已通过，2已拒绝 } ]， message:[ //仅考虑小规模数据，所以发送全部消息 { id: //聊天对象的id send:[ //你发送的所有消息 { data:, //消息内容 time: //unix时间戳，发送时间 } ], receive:[ //对方发送的所有消息 { data:, //消息内容 time: //unix时间戳，发送时间 readed: //数字表示是否阅读过。0未读，1已读 } ] } ] } 若登录失败： { ope:1, status:0, // info: //服务器生成的出错信息 } 登录窗口传递给好友窗口的data: { you:{ //你的个人信息 nick:, avatar:, id:, }， invite:[ //好友申请的数组 { from: //数字表示是你的申请还是他人对你申请 //0他人对你，1你对他人 id: nick: avatar:1 //数字代表固定头像 status: //0未审批，1已通过，2已拒绝 } ]， message:[ //仅考虑小规模数据，所以发送全部消息 { id: //聊天对象的id send:[ //你发送的所有消息 { data:, //消息内容 time: //unix时间戳，发送时间 } ], receive:[ //对方发送的所有消息 { data:, //消息内容 time: //unix时间戳，发送时间 readed: //数字表示是否阅读过。0未读，1已读 } ] } ] } 注册窗口： { ope:2, //一个数字代表操作 id:, pw:, avatar:, nick, } 服务器返回： { ope:2, status: //数字表示成功失败 info: //错误信息 } 聊天窗口： 用户点击发送消息： { ope:3, //操作代码 from:, //为方便服务器，发送者（你）的id to:, //接收者的id data:, //消息内容 time: //unix时间戳，发送时间 } 用户从服务器接收消息： { ope:3, // from:, // data:, //消息内容 time: //unix时间戳，发送时间 } 设置个人信息： { ope:4, // id:, pw:, avatar:, nick, } 加好友窗口： 点击搜索好友: { ope:5, data: //搜索框输入的关键字，可能是账号或昵称 } 服务器返回搜索结果： { ope:5, id:{ //如果有，返回id严格匹配的用户 avatar:, nick:, id:, }, nick:[ { avatar:, nick:, id:, }, {} ] } 点击加好友： { ope:6, from:, to: } 审核好友申请： 点击同意或拒绝（进行审核） { ope:7, from:, to:, status: } 从服务器接收好友申请结果 { ope:7, from:, status: } 客户端下线发给服务器消息 { ope:8, id:, //你的id } 客户端接收服务器消息更新好友在线状态 { ope:8, id:, //下线者id } 好友窗口传给聊天框的data： { }
